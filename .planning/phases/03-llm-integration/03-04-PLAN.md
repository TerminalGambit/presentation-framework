```python
@mcp.tool()
def optimize_slide(slide_yaml: str) -> dict:
    """Split an overflowing slide into multiple non-overflowing slides.

    Accepts a YAML string representing a single slide dict.
    Returns a dict with "slides" key containing a list of slide dicts,
    or "error" on failure.
    """
```

Implementation:
1. `import yaml` (already imported at top? check — if not, add to imports)
2. Parse `slide_yaml` with `yaml.safe_load(slide_yaml)` — wrap in try/except for invalid YAML
3. Import `split_slide` from `pf.optimizer`
4. Call `split_slide(slide_dict)`
5. Return `{"slides": result, "count": len(result), "was_split": len(result) > 1}`
6. On exception: return `{"error": str(e)}`

   ```python
   class SuggestionList(BaseModel):
   ```
